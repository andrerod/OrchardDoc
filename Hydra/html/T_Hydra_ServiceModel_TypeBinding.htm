Examples
The following example demonstrates binding an XmlElement to the Type defined by ExtensionConfiguration.
Copy
[XmlElement]
[TypeBinding]
public class ExtensionConfiguration
{...}
TypeBinding also allows you to specify the Type, which is useful if you wish to bind to a Type that contains only a subset of the values. In the following example, we only want the StoreName and Thumbprint, so specify the Type containing just those two properties.
Copy
public class StoredCertificateSettings
{
    [Required]
    public string StoreName { get; set; }
    [Required]
    public string Thumbprint { get; set; }
}
...
[XmlElement("CertificateSetting")]
[TypeBinding(typeof(StoredCertificateSettings))]
public class StoredCertificateSettingsXmlFormat
{
    [XmlElement(ConstantValue = "LocalMachine")]
    [PropertyBinding]
    public string StoreLocation { get; set; }
    [XmlElement]
    [PropertyBinding]
    public string StoreName { get; set; }
    [XmlElement]
    [PropertyBinding]
    public string Thumbprint { get; set; }
}
