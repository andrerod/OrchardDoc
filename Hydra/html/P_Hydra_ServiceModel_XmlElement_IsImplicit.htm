You can declare the parent element 'DedicatedCircuit', and then mark the child object with IsImplicit = true.
Copy
[XmlElement("DedicatedCircuit", XmlNamespace = "http://schemas.microsoft.com/windowsazure")]
[TypeBinding]
public class DedicatedCircuitGetResponse : OperationResponse
{
    [Documentation("Details of the requested dedicated circuit.")]
    [XmlElement(IsImplicit = true)]
    [PropertyBinding]
    public AzureDedicatedCircuit DedicatedCircuit { get; set; }
}
