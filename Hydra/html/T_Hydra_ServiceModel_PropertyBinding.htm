Examples
The following binds Label as a property of the 'parameters' parameter (defined as a ParameterBinding.) In the generated code, this allows you to specify the label of a cloud service using parameter.Label = "myCloud".
Copy
[ParameterBinding("parameters")]
public class CloudServiceCreateParameters
{
    [XmlElement]
    [Documentation("Label for the cloud service.")]
    [Required]
    [MaximumLength(100)]
    [PropertyBinding]
    public string Label { get; set; }
    ... }
